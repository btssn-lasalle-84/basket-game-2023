153 void Basketgame::chronometrerTour()
154 {
155  // Voir aussi : QElapsedTimer
156  QTime tempsEcoule = tempsTour->addSecs(-1);
157  tempsTour->setHMS(tempsEcoule.hour(),
158  tempsEcoule.minute(),
159  tempsEcoule.second());
160  // qDebug() << Q_FUNC_INFO << "tempsTours" <<
161  // tempsTours->toString("hh:mm:ss");
162  ui->tempsTour->setText(tempsTour->toString("hh:mm:ss"));
163 
164  if(*tempsTour == QTime(0, 0))
165  {
166  minuteurTour->stop();
167  initialiserDureeTour();
168  ui->tempsTour->setText(tempsTour->toString("hh:mm:ss"));
169  minuteurTour->start(TIC_HORLOGE);
170  qDebug() << Q_FUNC_INFO << "estEquipeRouge"
171  << plateau->estEquipeRouge();
172  if(plateau->estEquipeRouge())
173  {
174  ui->labelVisualisationEquipeRouge->setStyleSheet(
175  "background-color: transparent; color: black;");

281 void Basketgame::afficherUnJeton(int ligne, int colonne)
282 {
283  if(ligne < 0 || ligne >= NB_LIGNES)
284  return;
285  if(colonne < 0 || colonne >= NB_COLONNES)
286  return;
287 
288  QImage jetonRouge(JETON_ROUGE);
289  QImage jetonJaune(JETON_JAUNE);
290  QPixmap puissance4 = ui->labelVisualisationPlateau->pixmap()->copy();
291  QPainter p(&puissance4);
292 
293  qDebug() << Q_FUNC_INFO << "rouge" << plateau->estEquipeRouge() << "ligne"
294  << ligne << "colonne" << colonne;
295  if(plateau->estEquipeRouge())
296  {
297  p.drawImage(QPoint(DEPLACEMENT_X + (colonne * TAILLE_JETON),
298  DEPLACEMENT_Y - (ligne * TAILLE_JETON)),
299  jetonRouge);
300  p.end();
301  ui->labelVisualisationPlateau->setPixmap(puissance4);
302  }
303  else
304  {
305  p.drawImage(QPoint(DEPLACEMENT_X + (colonne * TAILLE_JETON),
306  DEPLACEMENT_Y - (ligne * TAILLE_JETON)),
307  jetonJaune);
308  p.end();
309  ui->labelVisualisationPlateau->setPixmap(puissance4);
310  }
311 }
