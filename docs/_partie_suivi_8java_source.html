747  @SuppressLint("HandlerLeak")
748  private void initialiserHandler()
749  {
750  Log.d(TAG, "initialiserHandler()");
751  handler = new Handler() {
752  public void handleMessage(Message message)
753  {
754  super.handleMessage(message);
755  // Log.d(TAG, "handleMessage() what = " + message.what);
756 
757  switch(message.what)
758  {
759  case CommunicationBluetooth.CONNEXION_BLUETOOTH:
760  Log.d(TAG, "handleMessage() CONNEXION_BLUETOOTH " + message.obj.toString());
761  afficherConnexionBluetooth(message.obj.toString());
762  gererConnexionBluetooth(message.obj.toString());
763  break;
764  case CommunicationBluetooth.RECEPTION_BLUETOOTH:
765  Log.d(TAG, "handleMessage() RECEPTION_BLUETOOTH " + message.obj.toString());
766  String trame = message.obj.toString();
767  if(verifierTrame(trame))
